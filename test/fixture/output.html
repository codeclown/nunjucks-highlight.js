function map(fn, a) {
    for (i = 0; i < a.length; i++) {
        a[i] = fn(a[i]);
    }
}

function reduce(fn, a, init) {
    var s = init;
    for (i = 0; i < a.length; i++) {
        s = fn( s, a[i] );
    }
    return s;
}
